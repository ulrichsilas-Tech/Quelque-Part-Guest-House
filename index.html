-- ════ ExamPlatform — Script SQL complet avec RLS ════ -- À exécuter dans Supabase → SQL Editor -- ══ 1. TABLES ══ create table if not exists admins ( id uuid primary key default gen_random_uuid(), identifiant text unique not null, nom text, email text, pwd text, role text default 'enseignant', valide boolean default false, created_at timestamptz default now() ); create table if not exists universites ( id uuid primary key default gen_random_uuid(), nom text not null, entites jsonb default '[]', logo_url text ); create table if not exists entites ( id uuid primary key default gen_random_uuid(), nom text not null, universite_id uuid references universites(id) on delete cascade, logo_url text ); create table if not exists departements ( id uuid primary key default gen_random_uuid(), nom text not null, entite_id uuid references entites(id) on delete cascade ); create table if not exists niveaux ( id uuid primary key default gen_random_uuid(), nom text not null, ordre int default 0, departement_id uuid references departements(id) on delete cascade ); create table if not exists filieres ( id uuid primary key default gen_random_uuid(), nom text not null, code text not null, ordre int default 0, departement_id uuid references departements(id) on delete cascade, niveau_id uuid references niveaux(id) on delete cascade ); create table if not exists parcours ( id uuid primary key default gen_random_uuid(), universite_id uuid references universites(id) on delete cascade, entite_id uuid references entites(id) on delete cascade, departement_id uuid references departements(id) on delete cascade, niveau_id uuid references niveaux(id) on delete cascade, admin_id uuid references admins(id) on delete cascade, created_at timestamptz default now() ); create table if not exists parcours_filieres ( parcours_id uuid references parcours(id) on delete cascade, filiere_id uuid references filieres(id) on delete cascade, primary key (parcours_id, filiere_id) ); create table if not exists exam_config ( id uuid primary key default gen_random_uuid(), admin_id uuid references admins(id) on delete cascade, parcours_id uuid references parcours(id) on delete set null, ecole text, institut text, matiere text, niveau text, annee text, date_exam date, heure time default '08:00', heure_fin time, duree int default 90, session text default 'Normale', bareme int default 20, consigne text, enseignant text, statut text default 'ferme', email_verification boolean default false, resend_key text, smtp_from text, reclamation_email text, logo_universite text, logo_entite text, -- pwd_admin text, (vestige supprime - utiliser admins.pwd) created_at timestamptz default now() ); create table if not exists candidats ( id uuid primary key default gen_random_uuid(), admin_id uuid references admins(id) on delete cascade, exam_config_id uuid references exam_config(id) on delete cascade, matricule text not null, nom text, prenom text, filiere text, filiere_id uuid references filieres(id) on delete set null, email text, date_naissance date, unique(exam_config_id, matricule) ); create table if not exists questions ( id uuid primary key default gen_random_uuid(), admin_id uuid references admins(id) on delete cascade, exam_config_id uuid references exam_config(id) on delete cascade, num int, ordre int, texte text, image_url text, html_content text ); create table if not exists sous_questions ( id text primary key, question_id uuid references questions(id) on delete cascade, ordre int, texte text, correct boolean default false ); create table if not exists copies ( id uuid primary key default gen_random_uuid(), admin_id uuid references admins(id) on delete cascade, exam_config_id uuid references exam_config(id) on delete cascade, matricule text, nom text, prenom text, filiere text, reponses jsonb, note numeric, note_brute numeric, note_sur_bareme numeric, date_depot timestamptz default now() ); create table if not exists tokens_composition ( token uuid primary key default gen_random_uuid(), matricule text, admin_id uuid, exam_config_id uuid references exam_config(id) on delete cascade, expires_at timestamptz, used boolean default false ); create table if not exists reset_tokens ( id uuid primary key default gen_random_uuid(), token text, admin_id uuid, expires_at timestamptz, used boolean default false ); create table if not exists invitations ( id uuid primary key default gen_random_uuid(), exam_config_id uuid references exam_config(id) on delete cascade, candidat_id uuid references candidats(id) on delete cascade, matricule text not null, email text not null, statut text default 'en_attente', erreur text, sent_at timestamptz, created_at timestamptz default now(), unique(exam_config_id, matricule) ); -- ══ 2. SUPER ADMIN PAR DÉFAUT ══ insert into admins (identifiant, nom, email, pwd, role, valide) values ('admin', 'Super Administrateur', 'admin@examplatform.com', 'admin123', 'superadmin', true) on conflict (identifiant) do nothing; -- ══ 3. GRANTs (obligatoire Supabase mai 2025+) ══ do $$ declare t text; tables text[] := array[ 'admins','exam_config','candidats','questions','sous_questions', 'copies','tokens_composition','reset_tokens','invitations', 'universites','entites','departements','niveaux','filieres', 'parcours','parcours_filieres' ]; begin foreach t in array tables loop execute format('grant select, insert, update, delete on public.%I to anon', t); execute format('grant select, insert, update, delete on public.%I to authenticated', t); execute format('grant select, insert, update, delete on public.%I to service_role', t); end loop; end $$; -- ══ 4. ACTIVATION RLS ══ do $$ declare t text; tables text[] := array[ 'admins','exam_config','candidats','questions','sous_questions', 'copies','tokens_composition','reset_tokens','invitations', 'universites','entites','departements','niveaux','filieres', 'parcours','parcours_filieres' ]; begin foreach t in array tables loop execute format('alter table public.%I enable row level security', t); end loop; end $$; -- ══ 5. POLICIES RLS ══ -- Architecture : pas d'auth Supabase, login custom via table admins. -- Stratégie : anon peut tout faire (l'app gère elle-même les droits), -- MAIS on bloque la colonne pwd dans admins pour la lecture externe. -- Tables publiques en lecture (hiérarchie universitaire — visible par étudiants) create policy "public_read" on universites for select to anon using (true); create policy "public_read" on entites for select to anon using (true); create policy "public_read" on departements for select to anon using (true); create policy "public_read" on niveaux for select to anon using (true); create policy "public_read" on filieres for select to anon using (true); create policy "public_read" on parcours for select to anon using (true); create policy "public_read" on parcours_filieres for select to anon using (true); create policy "public_read" on exam_config for select to anon using (true); create policy "public_read" on candidats for select to anon using (true); create policy "public_read" on questions for select to anon using (true); create policy "public_read" on sous_questions for select to anon using (true); create policy "public_read" on copies for select to anon using (true); create policy "public_read" on tokens_composition for select to anon using (true); create policy "public_read" on reset_tokens for select to anon using (true); create policy "public_read" on invitations for select to anon using (true); -- Écriture autorisée pour toutes les tables (l'app gère les droits) create policy "anon_write" on universites for all to anon using (true) with check (true); create policy "anon_write" on entites for all to anon using (true) with check (true); create policy "anon_write" on departements for all to anon using (true) with check (true); create policy "anon_write" on niveaux for all to anon using (true) with check (true); create policy "anon_write" on filieres for all to anon using (true) with check (true); create policy "anon_write" on parcours for all to anon using (true) with check (true); create policy "anon_write" on parcours_filieres for all to anon using (true) with check (true); create policy "anon_write" on exam_config for all to anon using (true) with check (true); create policy "anon_write" on candidats for all to anon using (true) with check (true); create policy "anon_write" on questions for all to anon using (true) with check (true); create policy "anon_write" on sous_questions for all to anon using (true) with check (true); create policy "anon_write" on copies for all to anon using (true) with check (true); create policy "anon_write" on tokens_composition for all to anon using (true) with check (true); create policy "anon_write" on reset_tokens for all to anon using (true) with check (true); create policy "anon_write" on invitations for all to anon using (true) with check (true); -- Table admins : lecture restreinte (pas de select * depuis l'extérieur) -- Seul le login (select par identifiant) est autorisé — la colonne pwd -- ne peut pas être masquée par RLS seul, donc on hache les mots de passe -- dans la prochaine étape (voir note ci-dessous) create policy "admins_read" on admins for select to anon using (true); create policy "admins_write" on admins for all to anon using (true) with check (true); -- ══ NOTE SÉCURITÉ ══ -- Les mots de passe dans admins sont en clair actuellement. -- Pour renforcer : activer le hachage bcrypt côté app (à faire en priorité). -- Protection immédiate : Supabase Dashboard → Settings → API → Allowed origins -- → ajouter uniquement l'URL de production (ex: https://votre-app.vercel.app) -- ══ 6. MIGRATION colonnes manquantes (projets existants) ══ alter table if exists entites add column if not exists logo_url text; alter table if exists universites add column if not exists logo_url text; alter table if exists copies add column if not exists note_brute numeric; alter table if exists copies add column if not exists note_sur_bareme numeric; alter table if exists sous_questions add column if not exists pts_v numeric default 0.25; alter table if exists sous_questions add column if not exists pts_f numeric default -0.25; alter table if exists sous_questions add column if not exists correcte boolean; update sous_questions set correcte = correct where correcte is null and correct is not null; -- ══ Migration v7 : lier questions/candidats/copies à exam_config ══ alter table if exists questions add column if not exists exam_config_id uuid references exam_config(id) on delete cascade; alter table if exists candidats add column if not exists exam_config_id uuid references exam_config(id) on delete cascade; alter table if exists copies add column if not exists exam_config_id uuid references exam_config(id) on delete cascade; -- Ajouter exam_config_id à tokens_composition si manquant alter table if exists tokens_composition add column if not exists exam_config_id uuid references exam_config(id) on delete cascade; -- Peupler exam_config_id pour les données existantes update questions q set exam_config_id = (select id from exam_config ec where ec.admin_id = q.admin_id limit 1) where q.exam_config_id is null; update candidats c set exam_config_id = (select id from exam_config ec where ec.admin_id = c.admin_id limit 1) where c.exam_config_id is null; update copies c set exam_config_id = (select id from exam_config ec where ec.admin_id = c.admin_id limit 1) where c.exam_config_id is null; -- Nouvelle contrainte unique candidats par épreuve (supprimer l'ancienne si elle existe) alter table if exists candidats drop constraint if exists candidats_admin_id_matricule_key; do $$ begin if not exists (select 1 from pg_constraint where conname = 'candidats_exam_config_id_matricule_key') then alter table candidats add constraint candidats_exam_config_id_matricule_key unique(exam_config_id, matricule); end if; end $$; -- ══ 7.
